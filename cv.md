function duplicateCount(text){
  let arr = [];
  let duplicateArr = []
  text.toLowerCase().split('').map(el => {
    if (!arr.includes(el)) {
        arr.push(el)
     } else {
       duplicateArr.push(el)
     }
  })
  return new Set(duplicateArr).size
}
